Mini Code Snippet

// pages/personas.jsx
import { createClient } from "@supabase/supabase-js";

export async function getServerSideProps() {
  const supabase = createClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL,
    process.env.SUPABASE_ANON_KEY
  );
  const { data: personas } = await supabase.from("personas").select("*");
  return { props: { personas } };
}

export default function Personas({ personas }) {
  return (
    <div className="container">
      <h2 className="text-accent">Persona List</h2>
      {personas.map(p => (
        <div key={p.id} className="card glass-card p-3 mb-3">
          <h5>{p.name}</h5>
          <p>{p.title}</p>
        </div>
      ))}
    </div>
  );
}

This example shows how to fetch live data from Supabase inside a Next.js page using server-side rendering — the same approach you’ll use when transitioning your playbooks to live, database-driven modules.
