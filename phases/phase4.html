Mini Code Snippet

# .github/workflows/deploy.yml
name: Build & Deploy
on:
  push:
    branches: [ "main" ]

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout Code
        uses: actions/checkout@v3

      - name: Setup Node.js
        uses: actions/setup-node@v3
        with:
          node-version: 20

      - name: Install Dependencies
        run: npm ci

      - name: Run Tests
        run: npm test

      - name: Build App
        run: npm run build

      - name: Deploy to Vercel
        run: npx vercel --prod --token=${{ secrets.VERCEL_TOKEN }}

This GitHub Actions workflow automates your entire pipeline — testing, building, and deploying your application whenever code is pushed to the main branch. It creates a professional, hands-free deployment loop aligned with modern DevOps standards.
